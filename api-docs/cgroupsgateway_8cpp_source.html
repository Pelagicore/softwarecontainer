34 ReturnCode CgroupsGateway::readConfigElement(const json_t *element)
35 {
36  std::string settingKey;
37  std::string settingValue;
38 
39  if (!read(element, "setting", settingKey)) {
40  log_error() << "Key \"setting\" either not a string or not in json configuration";
41  return ReturnCode::FAILURE;
42  }
43 
44  if (!read(element, "value", settingValue)) {
45  log_error() << "Key \"value\" either not a string or not in json configuration";
46  return ReturnCode::FAILURE;
47  }
48 
49  if (m_settings.count(settingKey) > 0) {
50  log_warning() << "setting '" << settingKey << "' is set more than once, may be problematic.";
51  }
52  m_settings.insert( std::pair<std::string, std::string>(settingKey, settingValue) );
53 
54  return ReturnCode::SUCCESS;
55 }

57 bool CgroupsGateway::activateGateway()
58 {
59  ReturnCode success = ReturnCode::FAILURE;
60  for (auto& setting: m_settings) {
61  success = getContainer()->setCgroupItem(setting.first, setting.second);
62  if (success != ReturnCode::SUCCESS) {
63  log_error() << "Error activating Cgroups Gateway, could not set cgroup item "
64  << setting.first << ": " << setting.second;
65  break;
66  }
67  }
68 
69  return success == ReturnCode::SUCCESS;
70 }
