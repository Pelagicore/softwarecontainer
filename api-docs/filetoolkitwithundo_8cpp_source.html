32 FileToolkitWithUndo::~FileToolkitWithUndo()
33 {
34  bool success = true;
35  // Clean up all created directories, files, and mount points
36 
37  while (!m_cleanupHandlers.empty()) {
38  CleanUpHandler *c = m_cleanupHandlers.top();
39  m_cleanupHandlers.pop();
40 
41  if (isError(c->clean())) {
42  success = false;
43  }
44 
45  delete c;
46  }
47 
48  if(!success) {
49  log_error() << "One or more cleanup handlers returned error status, please check the log";
50  }
51 }

66 ReturnCode FileToolkitWithUndo::createDirectory(const std::string &path)
67 {
68  if (isDirectory(path)) {
69  return ReturnCode::SUCCESS;
70  }
71 
72  if(isError(createParentDirectory(path))) {
73  log_error() << "Couldn't create parent directory for " << path;
74  return ReturnCode::FAILURE;
75  }
76 
77  if (mkdir(path.c_str(), S_IRWXU | S_IRWXG | S_IRWXO) == -1) {
78  log_error() << "Could not create directory " << path << " - Reason : " << strerror(errno);
79  return ReturnCode::FAILURE;
80  }
81 
82  m_cleanupHandlers.push(new DirectoryCleanUpHandler(path));
83  log_debug() << "Created directory " << path;
84 
85  return ReturnCode::SUCCESS;
86 }

102 ReturnCode FileToolkitWithUndo::bindMount(const std::string &src, const std::string &dst, bool readOnly, bool enableWriteBuffer)
103 {
104  unsigned long flags = MS_BIND;
105  std::string fstype;
106  const void *data = nullptr;
107  int mountRes;
108 
109  if (!existsInFileSystem(src)) {
110  log_error() << src << " does not exist on the host, can not bindMount";
111  return ReturnCode::FAILURE;
112  }
113 
114  log_debug() << "Bind-mounting " << src << " in " << dst << ", flags: " << flags;
115 
116  if(enableWriteBuffer) {
117  std::string upperDir = tempDir("/tmp/sc-bindmount-upper-XXXXXX");
118  std::string workDir = tempDir("/tmp/sc-bindmount-work-XXXXXX");
119  fstype.assign("overlay");
120 
121  std::ostringstream os;
122  os << "lowerdir=" << src << ",upperdir=" << upperDir << ",workdir=" << workDir;
123  data = os.str().c_str();
124 
125  log_debug() << "enableWriteBuffer, config: " << os.str();
126 
127  mountRes = mount("overlay", dst.c_str(), fstype.c_str(), flags, data);
128  } else {
129  mountRes = mount(src.c_str(), dst.c_str(), fstype.c_str(), flags, data);
130  }
131 
132  if (mountRes == 0) {
133  log_verbose() << "Bind-mounted folder " << src << " in " << dst;
134  m_cleanupHandlers.push(new MountCleanUpHandler(dst));
135  } else {
136  log_error() << "Could not mount into container: src=" << src
137  << " , dst=" << dst << " err=" << strerror(errno);
138  return ReturnCode::FAILURE;
139  }
140 
141  if (readOnly) {
142  flags = MS_REMOUNT | MS_RDONLY | MS_BIND;
143 
144  log_debug() << "Re-mounting read-only" << src << " in "
145  << dst << ", flags: " << flags;
146  mountRes = mount(src.c_str(), dst.c_str(), fstype.c_str(), flags, data);
147  if (mountRes != 0) {
148  // Failure
149  log_error() << "Could not re-mount " << src << " , read-only on "
150  << dst << " err=" << strerror(errno);
151  return ReturnCode::FAILURE;
152  }
153  }
154  return ReturnCode::SUCCESS;
155 }

157 ReturnCode FileToolkitWithUndo::overlayMount(
158  const std::string &lower,
159  const std::string &upper,
160  const std::string &work,
161  const std::string &dst)
162 {
163  std::string fstype = "overlay";
164  unsigned long flags = MS_BIND;
165 
166  if (isError(createDirectory(lower))
167  || isError(createDirectory(upper))
168  || isError(createDirectory(work)))
169  {
170  log_error() << "Failed to create lower/upper/work directory for overlayMount. lower=" <<
171  lower << ", upper=" << upper << ", work=" << work;
172  return ReturnCode::FAILURE;
173  }
174 
175  std::string mountoptions = StringBuilder() << "lowerdir=" << lower
176  << ",upperdir=" << upper
177  << ",workdir=" << work;
178 
179  int mountRes = mount("overlay", dst.c_str(), fstype.c_str(), flags, mountoptions.c_str());
180 
181  if (mountRes == 0) {
182  log_verbose() << "overlayMounted folder " << lower << " in " << dst;
183  m_cleanupHandlers.push(new MountCleanUpHandler(dst));
184  m_cleanupHandlers.push(new OverlaySyncCleanupHandler(upper, lower));
185  m_cleanupHandlers.push(new DirectoryCleanUpHandler(upper));
186  m_cleanupHandlers.push(new DirectoryCleanUpHandler(work));
187  } else {
188  log_error() << "Could not mount into container: upper=" << upper
189  << ",lower=" << lower
190  << ",work=" << work
191  << " at dst=" << dst << " err=" << strerror(errno);
192  return ReturnCode::FAILURE;
193  }
194 
195  return ReturnCode::SUCCESS;
196 }

239  m_cleanupHandlers.push(new FileCleanUpHandler(source));

242  log_error() << "Error creating symlink " << destination
