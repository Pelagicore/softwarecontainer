92 ReturnCode SoftwareContainer::preload()
93 {
94  log_debug() << "Initializing container";
95  if (isError(m_container->initialize())) {
96  log_error() << "Could not setup container for preloading";
97  return ReturnCode::FAILURE;
98  }
99 
100  log_debug() << "Creating container";
101  if (isError(m_container->create())) {
102  return ReturnCode::FAILURE;
103  }
104 
105  log_debug() << "Starting container";
106  ReturnCode result = m_container->start(&m_pcPid);
107  if (isError(result)) {
108  log_error() << "Could not start the container during preload";
109  return ReturnCode::FAILURE;
110  }
111 
112  log_debug() << "Started container with PID " << m_pcPid;
113  m_containerState.setValueNotify(ContainerState::PRELOADED);
114  return ReturnCode::SUCCESS;
115 }

117 ReturnCode SoftwareContainer::init()
118 {
119  validateContainerID();
120 
121  if (m_mainLoopContext->gobj() == nullptr) {
122  log_error() << "Main loop context must be set first !";
123  return ReturnCode::FAILURE;
124  }
125 
126  if (getContainerState() != ContainerState::PRELOADED) {
127  if (isError(preload())) {
128  log_error() << "Failed to preload container";
129  return ReturnCode::FAILURE;
130  }
131  }

168 pid_t SoftwareContainer::launchCommand(const std::string &commandLine)
169 {
170  log_debug() << "launchCommand called with commandLine: " << commandLine;
171  pid_t pid = INVALID_PID;
172  ReturnCode result = m_container->attach(commandLine, &pid);
173  if (isError(result)) {
174  log_error() << "Attach returned invalid pid, launchCommand fails";
175  return INVALID_PID;
176  }
177 
178  // TODO: Why do we shutdown as soon as one process exits?
179  addProcessListener(m_connections, pid, [&](pid_t pid, int returnCode) {
180  shutdown();
181  }, m_mainLoopContext);
182 
183  return pid;
184 }

192 void SoftwareContainer::setGatewayConfigs(const GatewayConfiguration &configs)
193 {
194  // Go through the active gateways and check if there is a configuration for it
195  // If there is, apply it.
196 
197  for (auto &gateway : m_gateways) {
198  std::string gatewayId = gateway->id();
199 
200  if (configs.count(gatewayId) != 0) {
201  std::string config = configs.at(gatewayId);
202  gateway->setConfig(config);
203  }
204  }
205 
206  for (auto &gateway : m_gateways) {
207  if (gateway->isConfigured()) {
208  gateway->activate();
209  }
210  }
211 
212  m_containerState.setValueNotify(ContainerState::READY);
213 }

220 ReturnCode SoftwareContainer::shutdown(unsigned int timeout)
221 {
222  log_debug() << "shutdown called"; // << logging::getStackTrace();
223  if(isError(shutdownGateways())) {
224  log_error() << "Could not shut down all gateways cleanly, check the log";
225  }
226 
227  if(isError(m_container->destroy(timeout))) {
228  log_error() << "Could not destroy the container during shutdown";
229  return ReturnCode::FAILURE;
230  }
231 
232  m_containerState.setValueNotify(ContainerState::TERMINATED);
233  return ReturnCode::SUCCESS;
234 }

236 ReturnCode SoftwareContainer::shutdownGateways()
237 {
238  ReturnCode status = ReturnCode::SUCCESS;
239  for (auto &gateway : m_gateways) {
240  if (gateway->isActivated()) {
241  if (!gateway->teardown()) {
242  log_warning() << "Could not tear down gateway cleanly: " << gateway->id();
243  status = ReturnCode::FAILURE;
244  }
245  }
246  }
247 
248  m_gateways.clear();
249  return status;
250 }
