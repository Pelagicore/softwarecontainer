32 ReturnCode DeviceNodeGateway::readConfigElement(const json_t *element)
33 {
34  DeviceNodeGateway::Device dev;
35 
36  if (!read(element, "name", dev.name)) {
37  log_error() << "Key \"name\" missing or not a string in json configuration";
38  return ReturnCode::FAILURE;
39  }
40 
41  read(element, "major", dev.major);
42  read(element, "minor", dev.minor);
43  read(element, "mode", dev.mode);
44 
45  const bool majorSpecified = dev.major.length() > 0;
46  const bool minorSpecified = dev.minor.length() > 0;
47 
48  if (majorSpecified | minorSpecified) {
49  if (!checkIsStrValid(dev.major, "Major version", "Major version must be specified when minor is.")
50  || !checkIsStrValid(dev.minor, "Minor version", "Minor version must be specified when major is.")
51  || !checkIsStrValid(dev.mode, "Mode", "Mode has to be specified when major and minor is specified.")) {
52  return ReturnCode::FAILURE;
53  }
54  }
55 
56  m_devList.push_back(dev);
57  return ReturnCode::SUCCESS;
58 }

76 bool DeviceNodeGateway::activateGateway()
77 {
78  if (m_devList.empty()) {
79  log_info() << "Activate was called when no devices has been configured.";
80  return false;
81  }
82 
83  for (auto &dev : m_devList) {
84  log_info() << "Mapping device " << dev.name;
85 
86  if (dev.major.length() != 0) {
87 
88  int majorVersion, minorVersion, mode;
89  parseInt(dev.major.c_str(), &majorVersion);
90  parseInt(dev.minor.c_str(), &minorVersion);
91  parseInt(dev.mode.c_str(), &mode);
92 
93  // mknod dev.name c dev.major dev.minor
94  pid_t pid = INVALID_PID;
95  getContainer()->executeInContainer([&] () {
96  return mknod(dev.name.c_str(), S_IFCHR | mode,
97  makedev(majorVersion, minorVersion));
98  }, &pid);
99 
100  if (waitForProcessTermination(pid) != 0) {
101  log_error() << "Failed to create device " << dev.name;
102  return false;
103  }
104  } else {
105  // No major & minor numbers specified => simply map the device from
106  // the host into the container
107  getContainer()->mountDevice(dev.name);
108 
109  if (dev.mode.length() != 0) {
110  const int mode = std::atoi(dev.mode.c_str());
111 
112  pid_t pid = INVALID_PID;
113  getContainer()->executeInContainer([&] () {
114  return chmod(dev.name.c_str(), mode);
115  }, &pid);
116 
117  if (waitForProcessTermination(pid) != 0) {
118  log_error() << "Could not 'chmod " << dev.mode
119  << "' the mounted device " << dev.name;
120  return false;
121  }
122  }
123  }
124  }
125 
126  m_state = GatewayState::ACTIVATED;
127  return true;
128 }
