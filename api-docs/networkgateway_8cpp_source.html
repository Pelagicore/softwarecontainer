40 ReturnCode NetworkGateway::readConfigElement(const json_t *element)
41 {
42  Entry e;
43  if (!read(element, "type", e.type)) {
44  log_error() << "No type specified in network config.";
45  return ReturnCode::FAILURE;
46  }
47 
48  if (e.type != "INCOMING" && e.type != "OUTGOING") {
49  log_error() << e.type << " is not a valid type ('INCOMING' or 'OUTGOING')";
50  return ReturnCode::FAILURE;
51  }
52 
53  int p;
54  if (!read(element, "priority", p)) {
55  log_error() << "No priority specified in network config.";
56  return ReturnCode::FAILURE;
57  }
58  e.priority = p;
59 
60  if (e.priority < 1) {
61  log_error() << "Priority can not be less than 1 but is " << e.priority;
62  return ReturnCode::FAILURE;
63  }
64 
65  const json_t *rules = json_object_get(element, "rules");
66 
67  if (rules == nullptr) {
68  log_error() << "No rules specified";
69  return ReturnCode::FAILURE;
70  }
71 
72  if (!json_is_array(rules)) {
73  log_error() << "Rules not specified as an array";
74  return ReturnCode::FAILURE;
75  }
76 
77  size_t ix;
78  json_t *val;
79  json_array_foreach(rules, ix, val) {
80  if (json_is_object(val)) {
81  if (isError(parseRule(val, e.rules))) {
82  log_error() << "Could not parse rule config";
83  return ReturnCode::FAILURE;
84  }
85  } else {
86  log_error() << "formatting of rules array is incorrect.";
87  return ReturnCode::FAILURE;
88  }
89  }
90 
91  std::string readTarget;
92  if (!read(element, "default", readTarget)) {
93  log_error() << "No default target specified or default target is not a string.";
94  return ReturnCode::FAILURE;
95  }
96 
97  e.defaultTarget = parseTarget(readTarget);
98  if (e.defaultTarget == Target::INVALID_TARGET) {
99  log_error() << "Default target '" << readTarget << "' is not a supported target.Invalid target.";
100  return ReturnCode::FAILURE;
101  }
102 
103  m_entries.push_back(e);
104 
105  // --- TEMPORARY WORKAROUND ---
106  // in the wait of activate() being rewritten
107  if (e.defaultTarget == Target::ACCEPT) {
108  m_internetAccess = true;
109  m_gateway = "10.0.3.1";
110  }
111  // ----------------------------
112  return ReturnCode::SUCCESS;
113 }

115 ReturnCode NetworkGateway::parseRule(const json_t *element, std::vector<Rule> &rules)
116 {
117  Rule r;
118  std::string target;
119  if (!read(element, "target", target)) {
120  log_error() << "Target not specified in the network config";
121  return ReturnCode::FAILURE;
122  }
123 
124  r.target = parseTarget(target);
125  if (r.target == Target::INVALID_TARGET) {
126  log_error() << target << " is not a valid target.";
127  return ReturnCode::FAILURE;
128  }
129 
130  if (!read(element, "host", r.host)) {
131  log_error() << "Host not specified in the network config.";
132  return ReturnCode::FAILURE;
133  }
134 
135  // Parsing different port formats
136  json_t *port = json_object_get(element, "port");
137  if (port != nullptr) {
138  parsePort(port, r.ports);
139  }
140  // If there were no port configured, leave the port list empty
141  // and assume that all ports should be considered in the rule.
142 
143  rules.push_back(r);
144  return ReturnCode::SUCCESS;
145 }

147 ReturnCode NetworkGateway::parsePort(const json_t *element, std::vector<unsigned int> &ports)
148 {
149  // Port formatted as single integer
150  if (json_is_integer(element)) {
151  int port = json_integer_value(element);
152  ports.push_back(port);
153 
154  // Port formatted as a string representing a range
155  } else if (json_is_string(element)) {
156  std::string portRange = json_string_value(element);
157 
158  const std::string::size_type n = portRange.find("-");
159  const std::string first = portRange.substr(0, n);
160  const std::string last = portRange.substr(n + 1);
161 
162  int startPort;
163  if (!parseInt(first.c_str(), &startPort)) {
164  log_error() << "Starting port in range " << portRange << "is not an integer.";
165  return ReturnCode::FAILURE;
166  }
167 
168  int endPort;
169  if (!parseInt(first.c_str(), &endPort)) {
170  log_error() << "End port in range " << portRange << "is not an integer.";
171  return ReturnCode::FAILURE;
172  }
173 
174  for (int i = startPort; i <= endPort; ++i) {
175  ports.push_back(i);
176  }
177 
178  // Port formatted as a list of integers
179  } else if (json_is_array(element)) {
180  size_t ix;
181  json_t *val;
182  json_array_foreach(element, ix, val) {
183  if (!json_is_integer(val)) {
184  log_error() << "Entry in port array is not an integer.";
185  return ReturnCode::FAILURE;
186  }
187 
188  int port = json_integer_value(element);
189  ports.push_back(port);
190  }
191  } else {
192  log_error() << "Rules specified in an invalid format";
193  return ReturnCode::FAILURE;
194  }
195  return ReturnCode::SUCCESS;
196 }

214 bool NetworkGateway::activateGateway()
215 {
216  if (!hasContainer()) {
217  log_error() << "activate was called on an EnvironmentGateway which has no associated container";
218  return false;
219  }
220 
221  if (m_gateway.size() != 0) {
222  log_debug() << "Default gateway set to " << m_gateway;
223  } else {
224  m_internetAccess = false;
225  log_debug() << "No gateway. Network access will be disabled";
226  }
227 
228  if ( !isBridgeAvailable() ) {
229  log_error() << "Bridge not available, expected gateway to be " << m_gateway;
230  return false;
231  }
232 
233  if (m_internetAccess) {
234  generateIP();
235  return up();
236  } else {
237  return down();
238  }
239 }

273 bool NetworkGateway::up()
274 {
275  if (m_interfaceInitialized) {
276  log_debug() << "Interface already configured";
277  return true;
278  }
279 
280  log_debug() << "Attempting to bring up eth0";
281  ReturnCode ret = executeInContainer([this] {
282  Netlink n;
283 
284  Netlink::LinkInfo iface;
285  if (isError(n.findLink("eth0", iface))) {
286  log_error() << "Could not find interface eth0 in container";
287  return 1;
288  }
289 
290  int ifaceIndex = iface.first.ifi_index;
291  if (isError(n.linkUp(ifaceIndex))) {
292  log_error() << "Could not bring interface eth0 up in container";
293  return 2;
294  }
295 
296  in_addr ip_addr;
297  inet_aton(ip().c_str(), &ip_addr);
298  return isSuccess(n.setIP(ifaceIndex, ip_addr, 24)) ? 0 : 3;
299  });
300 
301  if (isSuccess(ret)) {
302  m_interfaceInitialized = true;
303  return setDefaultGateway();
304  } else {
305  log_debug() << "Failed to bring up eth0";
306  return false;
307  }
308 }

337 bool NetworkGateway::isBridgeAvailable()
338 {
339  Netlink::LinkInfo iface;
340  if (isError(m_netlinkHost.findLink(BRIDGE_INTERFACE, iface))) {
341  log_error() << "Could not find " << BRIDGE_INTERFACE << " in the host";
342  }
343 
344  std::vector<Netlink::AddressInfo> addresses;
345  if (isError(m_netlinkHost.findAddresses(iface.first.ifi_index, addresses))) {
346  log_error() << "Could not fetch addresses for " << BRIDGE_INTERFACE << " in the host";
347  }
348 
349  return isSuccess(m_netlinkHost.hasAddress(addresses, AF_INET, m_gateway.c_str()));
350 }
