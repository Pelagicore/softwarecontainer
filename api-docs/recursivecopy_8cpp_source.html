46 int copyFile(const char* srcPath, const struct stat* sb, int typeflag)
47 {
48  std::string dstPath;
49  std::string src(srcPath);
50 
51  if (src.find(srcRoot) != std::string::npos) {
52  dstPath = dstRoot + src.erase(0, srcRoot.length());
53  } else {
54  dstPath = dstRoot + src;
55  }
56 
57  switch(typeflag) {
58  case FTW_D:
59  mkdir(dstPath.c_str(), sb->st_mode);
60  break;
61  case FTW_F:
62  std::ifstream src(srcPath, std::ios::binary);
63  std::ofstream dst(dstPath, std::ios::binary);
64  dst << src.rdbuf();
65  }
66  return 0;
67 }
